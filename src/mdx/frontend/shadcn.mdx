# Shadcn UI

## Installation

```bash
npx shadcn-ui@latest init
```

## Basic Setup

```typescript
// components.json
{
  "$schema": "https://ui.shadcn.com/schema.json",

    "config": "tailwind.config.ts",
    "css": "src/app/globals.css",

    "cssVariables": true
  }

```

## Tips and Best Practices

### Component Installation
- Install components individually as needed
- Keep component customizations in a separate directory
- Use the CLI for consistent installation

### Styling
- Use CSS variables for theming
- Customize components through Tailwind classes
- Maintain consistent styling patterns

### Organization
- Keep components in a dedicated directory
- Document component modifications
- Use proper component composition

### Form
```bash
npx shadcn-ui@latest add form
```

```tsx
import { zodResolver } from "@hookform/resolvers/zod"
import { useForm } from "react-hook-form"
import * as z from "zod"

import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@/components/ui/form"

const formSchema = z.object({
  username: z.string().min(2).max(50),
})

export function FormDemo() {
  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
  })

  return (
    <Form {...form}>
      <FormField
        control={form.control}
        name="username"
        render={({ field }) => (
          <FormItem>
            <FormLabel>Username</FormLabel>
            <FormControl>
              <Input {...field} />
            </FormControl>
            <FormMessage />
          </FormItem>
        )}
      />
    </Form>
  )
}
```
